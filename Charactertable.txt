	`charactergender` varchar(45) NOT NULL,
	`characterclass` varchar(45) NOT NULL,
	`characterrace` varchar(45) NOT NULL,
	`characterlevel` int NOT NULL,
	`exp` int NOT NULL,
	`background` varchar(45) NOT NULL,

	`dex` int NOT NULL,
	`constitution` int NOT NULL,
	`intelligence` int NOT NULL,
	`wisdom` int NOT NULL,

	`charisma` int NOT NULL,
	`inspiration` int NOT NULL,
	`skillsacrobatics` int NOT NULL,
	`skillsanimalhandling` int NOT NULL,
	`skillsarcana` int NOT NULL,
	`skillsathletics` int NOT NULL,
	`skillsdeception` int NOT NULL,
	`skillshistory` int NOT NULL,

	`skillsintimidation` int NOT NULL,
	`skillsinvestigation` int NOT NULL,
	`skillsmedicine` int NOT NULL,
	`skillsnature` int NOT NULL,
	`skillsperception` int NOT NULL,

	`skillsreligion` int NOT NULL,
	`skillssleightofhand` int NOT NULL,
	`skillsstealth` int NOT NULL,
	`skillssurvival` int NOT NULL,
	`age` int NOT NULL,
	`eyes` VARCHAR(45) NOT NULL,
